\begin{equation} \label{2TCM}
{C_{PET}}(t) = (1-V_B)\Big({\frac{K_1 k_2}{k_2+k_3}} e^{-(k_2+k_3)t} + {\frac{K_1 k_3}{k_2+k_3}}\Big)  C_{P}(t) + V_B C_{P}(t) \ \ \ \
